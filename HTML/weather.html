def _processData(df):
    
    #starttime weekday,day,hour
    df['weekday'] = df['Start Time'].apply(lambda s : s.weekday())
    df['day'] = df['Start Time'].apply(lambda s : str(s)[:10])
    df['hour'] = df['Start Time'].apply(lambda s : s.hour)
    print('')
    
     #
    def haversine(lon1, lat1, lon2, lat2): 
        """
        Calculate the great circle distance between two points 
        on the earth (specified in decimal degrees)
        """
        lon1, lat1, lon2, lat2 = map(radians, [lon1, lat1, lon2, lat2])
        # haversine
        dlon = lon2 - lon1 
        dlat = lat2 - lat1 
        a = sin(dlat/2)**2 + cos(lat1) * cos(lat2) * sin(dlon/2)**2
        c = 2 * asin(sqrt(a)) 
        r = 6371 # 
        return c * r * 1000
    
    df['start_end_distance'] = df.apply(lambda s : haversine(s['Start Station Longitude'],s['Start Station Latitude'],s['End Station Longitude'],
                                                             s['End Station Latitude']),axis = 1)
    print("")
    
    return df
